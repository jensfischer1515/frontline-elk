Logstash Indexer @ Frontline


input {							
  redis {
    host => "redis"
    type => "ishop_logstash"
    key => "ftl-app-ishop-logstash"
    data_type => "list"
    codec => json
    add_field => { "source_host" => "live" }
  }
}
				

filter {
  if [source_host] == "live" and [type] == "ishop_logstash" {
    mutate {
      rename => [ "ipaddr", "ip" ]
    }
  }
}
				

output {
  if [source_host] == "live" {
    elasticsearch_http {
      host => "elasticsearch"
    }
  }
}
